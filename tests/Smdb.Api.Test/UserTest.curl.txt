# SimpleMDB API CRUD+L Test Commands
# File: SimpleMDB_API_Test_cURL.txt
# Base URL
BASE_URL="http://localhost:8080/api/v1"

##################################################
# USERS CRUD + LIST
##################################################

# CREATE - Agregar un usuario
curl -X POST $BASE_URL/users \
     -H "Content-Type: application/json" \
     -d '{
           "username": "johndoe",
           "email": "johndoe@example.com",
           "password": "password123"
         }'

# READ - Obtener un usuario por ID
USER_ID=1
curl -X GET "$BASE_URL/users/$USER_ID"

# UPDATE - Actualizar un usuario existente
curl -X PUT "$BASE_URL/users/$USER_ID" \
     -H "Content-Type: application/json" \
     -d '{
           "username": "johndoe_updated",
           "email": "johndoe_updated@example.com",
           "password": "newpassword123"
         }'

# DELETE - Eliminar un usuario
curl -X DELETE "$BASE_URL/users/$USER_ID"

# LIST - Obtener todos los usuarios
curl -X GET $BASE_URL/users
